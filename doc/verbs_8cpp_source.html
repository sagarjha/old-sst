237 
238  // local connection data
239  struct cm_con_data_t local_con_data;
240  // remote connection data. Obtained via TCP
241  struct cm_con_data_t remote_con_data;
242  // this is used to ensure that host byte order is correct at each node
243  struct cm_con_data_t tmp_con_data;
244 
245  // just a dummy character
246  char temp_char;
247  char tQ[2] = { 'Q', 0 };
248 
249  union ibv_gid my_gid;
250  if (gid_idx >= 0) {

252  check_for_error(!rc, "ibv_query_gid failed, error code is " + std::to_string(errno));
253  } else {
254  memset(&my_gid, 0, sizeof my_gid);
255  }
256 
257  // exchange using TCP sockets info required to connect QPs
258  local_con_data.addr = htonll((uintptr_t) (char*) write_buf);
259  local_con_data.rkey = htonl(write_mr->rkey);
260  local_con_data.qp_num = htonl(qp->qp_num);

264  (char *) &local_con_data, (char *) &tmp_con_data);
265  check_for_error(sync_ret_code >= 0, "Could not exchange qp data in connect_qp");
266  remote_con_data.addr = ntohll(tmp_con_data.addr);
267  remote_con_data.rkey = ntohl(tmp_con_data.rkey);
268  remote_con_data.qp_num = ntohl(tmp_con_data.qp_num);
269  remote_con_data.lid = ntohs(tmp_con_data.lid);
270  memcpy(remote_con_data.gid, tmp_con_data.gid, 16);
271  // save the remote side attributes, we will need it for the post SR
272  remote_props = remote_con_data;
273 
274  // modify the QP to init
275  modify_qp_to_init();
276 
277  // modify the QP to RTR
278  modify_qp_to_rtr();
279 
280  // modify it to RTS
281  modify_qp_to_rts();
282 
283  // sync to make sure that both sides are in states that they can connect to prevent packet loss
284  // just send a dummy char back and forth
285  sync_ret_code = tcp::sock_sync_data(sock, 1, tQ, &temp_char);
286  check_for_error(!sync_ret_code, "Could not sync in connect_qp after qp transition to RTS state");
287 }

370 void verbs_poll_completion() {
371  struct ibv_wc wc;
372  unsigned long start_time_msec;
373  unsigned long cur_time_msec;
374  struct timeval cur_time;
375  int poll_result;
376 
377  // poll the completion for a while before giving up of doing it ..
378  gettimeofday(&cur_time, NULL);
379  start_time_msec = (cur_time.tv_sec * 1000) + (cur_time.tv_usec / 1000);
380  do {
381  poll_result = ibv_poll_cq(g_res->cq, 1, &wc);
382  gettimeofday(&cur_time, NULL);
383  cur_time_msec = (cur_time.tv_sec * 1000) + (cur_time.tv_usec / 1000);
384  } while ((poll_result == 0) && ((cur_time_msec - start_time_msec) < MAX_POLL_CQ_TIMEOUT));
385 
386  check_for_error(!(poll_result < 0), "Poll completion failed");
387  check_for_error(!(poll_result == 0), "Completion wasn't found in the CQ after timeout");
388  // check the completion status (here we don't care about the completion opcode)
389  if (wc.status != IBV_WC_SUCCESS) {
390  cout << "got bad completion with status: 0x%x, vendor syndrome: " << wc.status << ", " << wc.vendor_err;
391  }
392 }

402 void resources_create() {
403  struct ibv_device **dev_list = NULL;
404  struct ibv_device *ib_dev = NULL;
405  int i;
406  int cq_size = 0;
407  int num_devices;
408  int rc = 0;
409 
410  // get device names in the system
411  dev_list = ibv_get_device_list(&num_devices);
412  check_for_error(dev_list, "ibv_get_device_list failed; returned a NULL list");
413 
414  // if there isn't any IB device in host
415  check_for_error(num_devices, "NO RDMA device present");
416  // search for the specific device we want to work with
417  for (i = 0; i < num_devices; i++) {
418  if (!dev_name) {
419  dev_name = strdup(ibv_get_device_name(dev_list[i]));
420  }
421  if (!strcmp(ibv_get_device_name(dev_list[i]), dev_name)) {
422  ib_dev = dev_list[i];
423  break;
424  }
425  }
426  // if the device wasn't found in host
427  check_for_error(ib_dev, "No RDMA devices found in the host");
428  // get device handle
429  g_res->ib_ctx = ibv_open_device(ib_dev);
430  check_for_error(g_res->ib_ctx, "Could not open RDMA device");
431  // we are now done with device list, free it
432  ibv_free_device_list(dev_list);
433  dev_list = NULL;
434  ib_dev = NULL;
435  // query port properties
436  rc = ibv_query_port(g_res->ib_ctx, ib_port, &g_res->port_attr);
437  check_for_error(!rc, "Could not query port properties, error code is " + std::to_string(rc));
438 
439  // allocate Protection Domain
440  g_res->pd = ibv_alloc_pd(g_res->ib_ctx);
441  check_for_error(g_res->pd, "Could not allocate protection domain");
442 
443  // get the device attributes for the device
444  ibv_query_device(g_res->ib_ctx, &g_res->device_attr);
445 
446  // set to 1000 entries, we actually don't need more than the number of nodes
447  cq_size = 1000;
448  g_res->cq = ibv_create_cq(g_res->ib_ctx, cq_size, NULL, NULL, 0);
449  check_for_error(g_res->cq, "Could not create completion queue, error code is " + std::to_string(errno));
450 }
