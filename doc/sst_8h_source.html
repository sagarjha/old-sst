329 void SST_reads<Row>::detect() {
330  while (!thread_shutdown) {
331  // one time predicates need to be evaluated only until they become true
332  auto pred_it = predicates.one_time_predicates.begin();
333  while (pred_it != predicates.one_time_predicates.end()) {
334  if (pred_it->first(this) == true) {
335  for (auto func : pred_it->second) {
336  func(this);
337  }
338  // erase the predicate as it was just found to be true
339  pred_it = predicates.one_time_predicates.erase(pred_it);
340  } else {
341  pred_it++;
342  }
343  }
344 
345  // recurrent predicates are evaluated each time they are found to be true
346  for (pred_it = predicates.recurrent_predicates.begin();
347  pred_it != predicates.recurrent_predicates.end(); ++pred_it) {
348  if (pred_it->first(this) == true) {
349  for (auto func : pred_it->second) {
350  func(this);
351  }
352  }
353  }
354 
355  // transition predicates are only evaluated when they change from false to true
356  pred_it = predicates.transition_predicates.begin();
357  auto pred_state_it = predicates.transition_predicate_states.begin();
358  while (pred_it != predicates.transition_predicates.end()) {
359  //*pred_state_it is the previous state of the predicate at *pred_it
360  bool curr_pred_state = pred_it->first(this);
361  if (curr_pred_state == true && *pred_state_it == false) {
362  for (auto func : pred_it->second) {
363  func(this);
364  }
365  }
366  *pred_state_it = curr_pred_state;
367
